\begin{algorithm}[t]
\caption{Streaming RL Training with Playbook Context Updater}
\label{alg:stream-trainer}
\begin{algorithmic}[1]

\REQUIRE Student policy $\pi_\theta$; reward function $R$; teacher LLM $\mathcal{T}$; reflector LLM $\mathcal{M}_{\mathrm{refl}}$; curator LLM $\mathcal{M}_{\mathrm{cur}}$; streaming dataset $\mathcal{D}$; playbook concise frequency $f_c$; max inner updates $K$; early-stop threshold $\delta$; rollout samples per prompt $N$

\STATE Initialize playbook store $\mathcal{P} \leftarrow \{\}$ \COMMENT{maps example id $\to$ playbook state}
\STATE Initialize solution buffer $\mathcal{S} \leftarrow \{\}$
\STATE Initialize student playbook snapshot $\hat{\mathcal{P}} \leftarrow \{\}$

\FOR{each batch $\mathcal{B} = \{(x_i, y_i^*)\}_{i=1}^{B}$ from $\mathcal{D}$}
    \STATE $\bar{s} \leftarrow \textsc{None}$ \COMMENT{previous iteration scores for early stopping}
    \FOR{$k = 1, \ldots, K$}
        \STATE \textbf{// --- Stage 1: Student Rollout ---}
        \FOR{$i = 1, \ldots, B$}
            \IF{$\hat{\mathcal{P}}[i] \neq \emptyset$}
                \STATE $\tilde{x}_i \leftarrow \textsc{InjectPlaybook}(\hat{\mathcal{P}}[i],\; x_i)$ \COMMENT{augment prompt with student playbook}
            \ELSE
                \STATE $\tilde{x}_i \leftarrow x_i$
            \ENDIF
        \ENDFOR
        \STATE Sample $\{y_i^{(n)}\}_{n=1}^{N} \sim \pi_\theta(\cdot \mid \tilde{x}_i)$ for each $i$ \COMMENT{$N$-sample rollout}

        \STATE \textbf{// --- Stage 2: Reward \& Feedback Collection ---}
        \STATE $r_i^{(n)},\; f_i^{\text{env}} \leftarrow R(x_i, y_i^{(n)}, y_i^*)$ for all $i, n$ \COMMENT{reward + environment feedback}
        \STATE $a_i \leftarrow \mathbf{1}[\max_n r_i^{(n)} \geq \tau]$ \COMMENT{per-example correctness}
        \IF{teacher $\mathcal{T}$ is enabled}
            \STATE $f_i^{\text{teach}} \leftarrow \mathcal{T}(x_i, y_i^{(1)}, f_i^{\text{env}})$ for incorrect $i$ \COMMENT{async teacher feedback}
        \ENDIF

        \STATE \textbf{// --- Stage 3: Playbook Context Update ---}
        \IF{context updater is enabled}
            \STATE \textbf{// Step 3a: Playbook Concising}
            \FOR{each active playbook key $p$}
                \IF{$\mathcal{P}[p].\text{count} \bmod f_c = 0$ \OR $|\mathcal{P}[p]| > B_{\max}$}
                    \STATE $\mathcal{P}[p] \leftarrow \textsc{Concise}(\mathcal{P}[p])$ \COMMENT{remove harmful/stale bullets}
                \ENDIF
            \ENDFOR

            \STATE \textbf{// Step 3b: Reflector Pass (incorrect samples)}
            \STATE $\mathcal{I} \leftarrow \{i : a_i = 0\}$ \COMMENT{incorrect sample indices}
            \FOR{$i \in \mathcal{I}$}
                \STATE $\rho_i \leftarrow \mathcal{M}_{\mathrm{refl}}(x_i,\; y_i^{(1)},\; f_i^{\text{env}},\; f_i^{\text{teach}},\; \mathcal{P}[i])$ \COMMENT{generate reflection}
                \STATE Extract bullet tags $\{(\text{id}, \text{helpful/harmful})\}$ from $\rho_i$
                \STATE Update bullet counts in $\mathcal{P}[i]$ accordingly
            \ENDFOR

            \STATE \textbf{// Step 3c: Success Tagging (correct samples)}
            \STATE $\mathcal{C} \leftarrow \{i : a_i = 1\}$
            \FOR{$i \in \mathcal{C}$}
                \STATE Run success reflector to tag helpful bullets in $\mathcal{P}[i]$
            \ENDFOR

            \STATE \textbf{// Step 3d: Curator Pass (incorrect samples)}
            \FOR{$i \in \mathcal{I}$}
                \STATE $\mathcal{O}_i \leftarrow \mathcal{M}_{\mathrm{cur}}(\rho_i,\; \textsc{Stats}(\mathcal{P}[i]),\; \mathcal{P}[i])$ \COMMENT{propose add/edit/remove ops}
                \STATE $\mathcal{P}[i] \leftarrow \textsc{ApplyOps}(\mathcal{P}[i],\; \mathcal{O}_i)$ \COMMENT{update playbook}
            \ENDFOR

            \STATE \textbf{// Step 3e: Sync Student Playbook Snapshot}
            \IF{sync condition met}
                \STATE $\hat{\mathcal{P}} \leftarrow \textsc{Copy}(\mathcal{P})$ \COMMENT{freeze snapshot for next student rollout}
            \ENDIF
        \ENDIF

        \STATE \textbf{// --- Stage 4: Teacher Prompt Construction ---}
        \FOR{$i = 1, \ldots, B$}
            \STATE $d_i \leftarrow$ successful demonstration from batch or $\mathcal{S}[i]$ \COMMENT{solution lookup}
            \STATE $\hat{x}_i \leftarrow \textsc{BuildTeacherPrompt}(x_i,\; \mathcal{P}[i],\; \rho_i,\; d_i,\; f_i^{\text{env}},\; f_i^{\text{teach}})$
        \ENDFOR

        \STATE \textbf{// --- Stage 5: Policy Update ---}
        \STATE Compute $\log \pi_\theta(y_i^{(n)} \mid x_i)$ and $\log \pi_{\text{ref}}(y_i^{(n)} \mid x_i)$ for all $i, n$
        \STATE Compute advantages $\hat{A}_i^{(n)}$ via GRPO: $\hat{A}_i^{(n)} = \frac{r_i^{(n)} - \text{mean}_n(r_i^{(\cdot)})}{\text{std}_n(r_i^{(\cdot)})}$
        \STATE Compute teacher log-likelihood $\log \pi_\theta(y_i^{(1)} \mid \hat{x}_i)$ using enriched prompt
        \STATE $\mathcal{L}_{\text{RL}} = -\frac{1}{BN}\sum_{i,n} \min\!\Big(\frac{\pi_\theta}{\pi_{\text{old}}} \hat{A}_i^{(n)},\; \text{clip}\big(\frac{\pi_\theta}{\pi_{\text{old}}},\; 1\!\pm\!\epsilon\big) \hat{A}_i^{(n)}\Big)$
        \STATE $\mathcal{L}_{\text{SD}} = -\frac{1}{|\{i : m_i\!=\!1\}|}\sum_{i: m_i=1} \log \pi_\theta(y_i^{(1)} \mid \hat{x}_i)$ \COMMENT{self-distillation on reprompted samples}
        \STATE $\theta \leftarrow \theta - \eta\,\nabla_\theta\!\big(\mathcal{L}_{\text{RL}} + \lambda\,\mathcal{L}_{\text{SD}}\big)$

        \STATE \textbf{// --- Early Stopping Check ---}
        \STATE $s_i \leftarrow \frac{1}{N}\sum_n r_i^{(n)}$ for all $i$
        \IF{$k \geq k_{\min}$ \AND $\frac{1}{B}\sum_i \mathbf{1}[s_i > \bar{s}_i] < \delta$}
            \STATE \textbf{break} \COMMENT{stop inner loop if insufficient improvement}
        \ENDIF
        \STATE $\bar{s} \leftarrow s$
        \STATE Sync $\pi_\theta$ weights to rollout workers

    \ENDFOR
    \STATE Cache successful solutions into $\mathcal{S}$
\ENDFOR

\end{algorithmic}
\end{algorithm}
